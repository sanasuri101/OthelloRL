# Eval Script + GitHub CI Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a fixed-opponent evaluation module that benchmarks the policy against negamax at multiple depths (logged to wandb during training and usable standalone), and set up GitHub remote + CI that rebuilds the C extension and runs the test suite on every push.

**Architecture:** `othello/eval.py` exposes `evaluate(policy, depths, n_games, device) -> dict[int, float]`; `train.py` calls it every `eval_interval` steps and logs `eval/win_rate_vs_negamax_d{depth}` to wandb. The Makefile is made cross-platform (macOS/Linux) to support GitHub Actions CI on `ubuntu-latest`.

**Tech Stack:** Python 3.12, PyTorch, numpy, uv, pytest, GitHub Actions, wandb (optional in eval CLI)

```python
"""Tests for othello/eval.py — fixed-opponent evaluation."""

import numpy as np
import pytest
import torch

from othello.othello import Othello
from othello.eval import evaluate


def _make_dummy_policy(obs_size: int = 131, act_size: int = 65, hidden_size: int = 64):
    """Tiny policy that outputs uniform logits — for structural testing only."""
    import torch.nn as nn

    class DummyPolicy(nn.Module):
        def __init__(self):
            super().__init__()
            self.lstm_cell = nn.LSTMCell(hidden_size, hidden_size)
            self.encoder = nn.Linear(obs_size, hidden_size)
            self.actor = nn.Linear(hidden_size, act_size)
            self.critic = nn.Linear(hidden_size, 1)

        def forward_eval(self, obs, state):
            h = state["lstm_h"].squeeze(0)
            c = state["lstm_c"].squeeze(0)
            enc = torch.relu(self.encoder(obs.float()))
            h, c = self.lstm_cell(enc, (h, c))
            state["lstm_h"] = h.unsqueeze(0)
            state["lstm_c"] = c.unsqueeze(0)
            return self.actor(h), self.critic(h)

    return DummyPolicy()


class TestEvaluate:
    def test_returns_dict_keyed_by_depth(self):
        """evaluate() returns a dict with one key per requested depth."""
        policy = _make_dummy_policy()
        results = evaluate(policy, depths=[1], n_games=5, device="cpu")
        assert isinstance(results, dict)
        assert 1 in results

    def test_win_rate_is_between_zero_and_one(self):
        """Win rate is a valid probability."""
        policy = _make_dummy_policy()
        results = evaluate(policy, depths=[1], n_games=10, device="cpu")
        assert 0.0 <= results[1] <= 1.0

    def test_multiple_depths_all_present(self):
        """All requested depths appear in results."""
        policy = _make_dummy_policy()
        results = evaluate(policy, depths=[1, 2], n_games=5, device="cpu")
        assert set(results.keys()) == {1, 2}

    def test_n_games_respected(self):
        """Each depth evaluates exactly n_games games."""
        policy = _make_dummy_policy()
        # Use a very small n_games to keep test fast
        results = evaluate(policy, depths=[1], n_games=3, device="cpu")
        # If n_games was respected, win_rate is a multiple of 1/3 (0, 1/3, 2/3, or 1)
        wr = results[1]
        assert any(abs(wr - k / 3) < 1e-9 for k in range(4))
```

### Task 3: Implement eval.py

**Files:**
- Create: `othello/eval.py`

- [ ] **Step 3.1: Create othello/eval.py**

```python
"""Fixed-opponent evaluation for Othello RL policies.

Usage (standalone):
    python othello/eval.py --checkpoint experiments/othello_ppo/checkpoint_000200.pt
    python othello/eval.py --checkpoint model.pt --depths 1,2,3,5 --n_games 200 --wandb
"""

from __future__ import annotations

import argparse
import sys
from pathlib import Path
from typing import Optional

import numpy as np
import torch
import torch.nn as nn


def evaluate(
    policy: nn.Module,
    depths: list[int] | None = None,
    n_games: int = 100,
    device: str = "cpu",
) -> dict[int, float]:
    """Evaluate *policy* against negamax at each depth in *depths*.

    Parameters
    ----------
    policy:
        An ``OthelloPolicy`` (or compatible) instance with a ``forward_eval``
        method and an ``lstm_cell`` attribute (used to infer hidden size).
    depths:
        Negamax search depths to evaluate against.  Defaults to ``[1, 2, 3, 5]``.
    n_games:
        Number of games to play at each depth.
    device:
        Torch device string (``"cpu"`` or ``"cuda"``).

    Returns
    -------
    dict[int, float]
        Win rate (0.0–1.0) for each depth.
    """
    from othello.othello import Othello  # local import — avoids circular at module level

    if depths is None:
        depths = [1, 2, 3, 5]

    hidden_size: int = policy.lstm_cell.hidden_size
    policy = policy.to(device)
    policy.eval()

    results: dict[int, float] = {}

    for depth in depths:
        env = Othello(num_envs=1)
        obs_np, _ = env.reset()

        lstm_state: dict[str, torch.Tensor] = {
            "lstm_h": torch.zeros(1, 1, hidden_size, device=device),
            "lstm_c": torch.zeros(1, 1, hidden_size, device=device),
        }

        games_done = 0
        wins = 0

        while games_done < n_games:
            obs_t = torch.tensor(obs_np, dtype=torch.float32, device=device)
            with torch.no_grad():
                logits, _ = policy.forward_eval(obs_t, lstm_state)
            action = logits.argmax(dim=-1).cpu().numpy().astype(np.int32)

            obs_np, rew_np, term_np, _trunc_np, _infos = env.step_negamax(
                action, depth
            )

            if term_np[0]:
                games_done += 1
                if rew_np[0] > 0:
                    wins += 1
                # Reset LSTM state at episode boundary
                lstm_state["lstm_h"].zero_()
                lstm_state["lstm_c"].zero_()

        env.close()
        results[depth] = wins / n_games

    return results


def _print_results(results: dict[int, float]) -> None:
    print("\nEvaluation Results")
    print("-" * 32)
    for depth, wr in sorted(results.items()):
        bar = "█" * int(wr * 20)
        print(f"  vs negamax d{depth}: {wr:.1%}  {bar}")
    print()


def main(argv: list[str] | None = None) -> None:
    parser = argparse.ArgumentParser(description="Evaluate an Othello RL checkpoint")
    parser.add_argument("--checkpoint", required=True, help="Path to .pt checkpoint file")
    parser.add_argument("--depths", default="1,2,3,5", help="Comma-separated negamax depths")
    parser.add_argument("--n_games", type=int, default=100, help="Games per depth")
    parser.add_argument("--device", default="cpu", help="Torch device")
    parser.add_argument("--wandb", action="store_true", help="Log results to wandb")
    parser.add_argument("--wandb_project", default="othello-rl")
    parser.add_argument("--wandb_run_name", default=None)
    args = parser.parse_args(argv)

    depths = [int(d.strip()) for d in args.depths.split(",")]
    ckpt_path = Path(args.checkpoint)
    if not ckpt_path.exists():
        print(f"Error: checkpoint not found: {ckpt_path}", file=sys.stderr)
        sys.exit(1)

    # ---- Load policy ------------------------------------------------
    # Import here so the module is usable without train.py on the path
    sys.path.insert(0, str(Path(__file__).parent.parent))
    from othello.train import make_policy  # noqa: PLC0415
    from othello.othello import Othello   # noqa: PLC0415

    _env = Othello(num_envs=1)
    policy = make_policy(_env, hidden_size=256).to(args.device)
    _env.close()

    ckpt = torch.load(args.checkpoint, map_location=args.device, weights_only=True)
    policy.load_state_dict(ckpt["policy_state_dict"])
    global_step = ckpt.get("global_step", 0)
    print(f"Loaded checkpoint: {ckpt_path}  (global_step={global_step})")

    # ---- Evaluate ---------------------------------------------------
    results = evaluate(policy, depths=depths, n_games=args.n_games, device=args.device)
    _print_results(results)

    # ---- Log to wandb -----------------------------------------------
    if args.wandb:
        import wandb

        run = wandb.init(
            project=args.wandb_project,
            name=args.wandb_run_name or f"eval_{ckpt_path.stem}",
            config={"checkpoint": str(ckpt_path), "n_games": args.n_games},
        )
        wandb.log(
            {f"eval/win_rate_vs_negamax_d{d}": wr for d, wr in results.items()},
            step=global_step,
        )
        run.finish()
        print("Results logged to wandb.")


if __name__ == "__main__":
    main()
```

**Files:**

The CI workflow must:
1. Install `uv` (Python package manager)
2. Sync deps via `uv sync`
3. Install `libraylib-dev` (raylib is required by the C extension)
4. Compile the C extension via `make`
5. Run pytest

Create `.github/workflows/ci.yml`:

```yaml
name: CI

on:
  push:
    branches: [master]
  pull_request:
    branches: [master]

jobs:
  test:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: Install uv
        uses: astral-sh/setup-uv@v4
        with:
          version: "latest"

      - name: Install Python deps
        working-directory: othello
        run: uv sync

      - name: Install raylib (required by C extension)
        run: sudo apt-get update && sudo apt-get install -y libraylib-dev

      - name: Build C extension
        working-directory: othello
        run: make PYTHON="$(uv run python -c 'import sys; print(sys.executable)')"

      - name: Run tests
        working-directory: othello
        run: uv run python -m pytest tests/ -v
```
